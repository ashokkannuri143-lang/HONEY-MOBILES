    <!-- Interactive Logic Execution -->
    <script>
        // Render beautiful vector icons dynamically
        lucide.createIcons();

        let cart = [];

        function toggleCart() {
            const drawer = document.getElementById('cart-drawer');
            drawer.classList.toggle('translate-x-full');
        }

        function addToCart(name, price) {
            cart.push({ name, price });
            updateCartUI();
            
            // Auto open drawer to visually notify user
            const drawer = document.getElementById('cart-drawer');
            if (drawer.classList.contains('translate-x-full')) {
                drawer.classList.remove('translate-x-full');
            }
        }

        function updateCartUI() {
            // Update counter pill
            document.getElementById('cart-count').innerText = cart.length;

            const itemsContainer = document.getElementById('cart-items');
            const totalContainer = document.getElementById('cart-total');

            if (cart.length === 0) {
                itemsContainer.innerHTML = 'Your cart feels light! Add some devices to get started.';
                totalContainer.innerText = '$0';
                return;
            }

            // Build structural product list inside drawer
            itemsContainer.innerHTML = cart.map((item, idx) => `
                <div class="flex justify-between items-center bg-gray-50 p-3 rounded-xl border border-gray-100">
                    <div>
                        <p class="font-bold text-gray-900">${item.name}</p>
                        <p class="text-xs text-amber-600 font-semibold">$${item.price}</p>
                    </div>
                    <button onclick="removeFromCart(${idx})" class="text-red-400 hover:text-red-600 transition p-1">
                        <i data-lucide="trash-2" class="w-4 h-4"></i>
                    </button>
                </div>
            `).join('');

            // Recalculate dynamic totals
            const total = cart.reduce((sum, item) => sum + item.price, 0);
            totalContainer.innerText = `$${total}`;
            
            // Re-render icons created inside template strings
            lucide.createIcons();
        }

        function removeFromCart(index) {
            cart.splice(index, 1);
            updateCartUI();
        }

        function handleContact(e) {
            e.preventDefault();
            alert('Thank you! Your inquiry was submitted to Honey Mobiles.');
            e.target.reset();
        }
    </script>
